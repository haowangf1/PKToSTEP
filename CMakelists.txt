cmake_minimum_required(VERSION 3.15)
project(PKToSTEP LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(AMCAXExchangeBase REQUIRED)
find_package(pskernel REQUIRED)
find_package(AMXT_STP REQUIRED)
find_package(GTest REQUIRED)

# 主程序
add_executable(PKToSTEP
    src/main.cpp
    src/pk_to_xchg_converter.cpp
)

target_include_directories(PKToSTEP PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_link_libraries(PKToSTEP PRIVATE
    AMCAXExchangeBase::AMCAXExchangeBase
    pskernel::pskernel
    AMXT_STP::AMXT_STP
)

# 启用测试
enable_testing()

# 测试工具库
add_library(test_utils
    test/test_utils.cpp
)

target_include_directories(test_utils PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}/test
)

target_link_libraries(test_utils PUBLIC
    AMCAXExchangeBase::AMCAXExchangeBase
    pskernel::pskernel
    AMXT_STP::AMXT_STP
)

# 往返测试
add_executable(test_roundtrip
    test/test_roundtrip.cpp
    src/pk_to_xchg_converter.cpp
)

target_include_directories(test_roundtrip PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}/test
)

target_link_libraries(test_roundtrip PRIVATE
    test_utils
    GTest::gtest
    GTest::gtest_main
    AMCAXExchangeBase::AMCAXExchangeBase
    pskernel::pskernel
    AMXT_STP::AMXT_STP
)

# 添加到 CTest
add_test(NAME RoundtripTest COMMAND test_roundtrip)
