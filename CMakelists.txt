cmake_minimum_required(VERSION 3.15)
project(PKToSTEP LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(AMCAXExchangeBase REQUIRED)
find_package(pskernel REQUIRED)
find_package(AMXT_STP REQUIRED)
find_package(GTest REQUIRED)

add_executable(PKToSTEP
    src/main.cpp
    src/pk_to_xchg_converter.cpp
)

target_include_directories(PKToSTEP PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_link_libraries(PKToSTEP PRIVATE
    AMCAXExchangeBase::AMCAXExchangeBase
    pskernel::pskernel
    AMXT_STP::AMXT_STP
)

# Tests
enable_testing()

add_executable(test_roundtrip
    test/test_roundtrip.cpp
    src/pk_to_xchg_converter.cpp
)

target_include_directories(test_roundtrip PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_compile_definitions(test_roundtrip PRIVATE
    PROJECT_ROOT_DIR="${CMAKE_CURRENT_SOURCE_DIR}"
)

target_link_libraries(test_roundtrip PRIVATE
    AMCAXExchangeBase::AMCAXExchangeBase
    pskernel::pskernel
    AMXT_STP::AMXT_STP
    GTest::gtest
    GTest::gtest_main
)

add_test(NAME RoundtripTests COMMAND test_roundtrip)
